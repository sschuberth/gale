98  void fromRGB(double const r,double const g,double const b) {
99  double m=min(r,g,b);
100  double V=max(r,g,b);
101 
102  double delta=V-m;
103 
104  double H,S;
105 
106  if (delta) {
107  // If delta>0, it is V>0, too.
108  S=delta/V;
109 
110  if (r==V) {
111  H=(g-b)/delta;
112  }
113  else if (g==V) {
114  H=(b-r)/delta+2;
115  }
116  else {
117  H=(r-g)/delta+4;
118  }
119  }
120  else {
121  // Achromatic case (gray), actually hue is undefined!
122  H=S=0;
123  }
124 
125  // Map H to range [0,360[ and SV to range [0,100].
126  m_h=wrap(H*60.0,360.0);
127  m_s=S*100.0;
128  m_v=V*100.0;
129  }

133  void toRGB(double& r,double& g,double& b) const {
134  // Map H to range [0,6[ and SV to range [0,1].
135  double H=wrap(m_h/60.0,6.0);
136  double S=clamp(m_s/100.0,0.0,1.0);
137  double V=clamp(m_v/100.0,0.0,1.0);
138 
139  // Convert to RGB.
140  if (S) {
141  int i=static_cast<int>(roundToZero(H));
142  double f=H-i;
143 
144  double p=V*(1-S);
145  double q=V*(1-S*f);
146  double t=V*(1-S*(1-f));
147 
148  switch (i) {
149  default: {
150  r=V; g=t; b=p;
151  break;
152  }
153  case 1: {
154  r=q; g=V; b=p;
155  break;
156  }
157  case 2: {
158  r=p; g=V; b=t;
159  break;
160  }
161  case 3: {
162  r=p; g=q; b=V;
163  break;
164  }
165  case 4: {
166  r=t; g=p; b=V;
167  break;
168  }
169  case 5: {

171  break;
172  }
173  }
174  }
