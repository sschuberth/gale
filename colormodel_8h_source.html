93  static void RYBHueToRGBHue(double const h_ryb,double& h_rgb) {
94  ColorSpaceHSV hsv(h_ryb,100,100);
95 
96  // Abuse the HSV to RGB conversion: The amount of green for a given hue
97  // in RGB matches the amount of yellow for the same hue in RYB.
98  double r,y,b;

114  static void RGBHueToRYBHue(double const h_rgb,double& h_ryb,double const a=0,double const b=360) {
115  // Guess an RYB hue ...
116  h_ryb=(a+b)*0.5;
117 
118  // ... and convert it to an RGB hue ...
119  double h_rgb_guessed;
120  RYBHueToRGBHue(h_ryb,h_rgb_guessed);
121 
122  // ... to see how closely we match the given RGB hue ...
123  double h_rgb_diff=h_rgb_guessed-h_rgb;
124  if (abs(h_rgb_diff)<=Numd::ZERO_TOLERANCE()) {
125  return;
126  }
127  // ... and adjust our guess accordingly, if required.
128  else if (h_rgb_diff<0) {
129  RGBHueToRYBHue(h_rgb,h_ryb,h_ryb,b);
130  }
131  else {
132  RGBHueToRYBHue(h_rgb,h_ryb,a,h_ryb);
133  }
134  }

157  void toRGB(double& r,double& g,double& b) const {
158  // Pair-wise interpolate RGB corner values along the B-axis of the RYB
159  // interpolation cube.
160  Tup3d b00=cubic(WHITE() , BLUE() , m_b);
161  Tup3d b10=cubic(YELLOW() , GREEN() , m_b);
162  Tup3d b01=cubic(RED() , PURPLE() , m_b);
163  Tup3d b11=cubic(ORANGE() , BLACK() , m_b);
164 
165  // Pair-wise interpolate RGB edge values along the Y-axis of the RYB
166  // interpolation cube.
167  Tup3d y0=cubic(b00,b10,m_y);
168  Tup3d y1=cubic(b01,b11,m_y);
169 
170  // Pair-wise interpolate RGB face values along the R-axis of the RYB
171  // interpolation cube.
172  Tup3d rgb=cubic(y0,y1,m_r);
173 
174  r=rgb[0];
175  g=rgb[1];
176  b=rgb[2];
177  }
