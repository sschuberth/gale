99  void fromRGB(double const r,double const g,double const b) {
100  double m=min(r,g,b);
101  double V=max(r,g,b);
102 
103  double delta=V-m;
104 
105  double H,S;
106 
107  if (delta) {
108  // If delta>0, it is V>0, too.
109  S=delta/V;
110 
111  if (r==V) {
112  H=(g-b)/delta;
113  }
114  else if (g==V) {
115  H=(b-r)/delta+2;
116  }
117  else {
118  H=(r-g)/delta+4;
119  }
120  }
121  else {
122  // Achromatic case (gray), actually hue is undefined!
123  H=S=0;
124  }
125 
126  // Map H to range [0,360[ and SV to range [0,100].
127  m_h=wrap(H*60.0,360.0);
128  m_s=S*100.0;
129  m_v=V*100.0;
130  }
131 
134  void toRGB(double& r,double& g,double& b) const {
135  // Map H to range [0,6[ and SV to range [0,1].
136  double H=wrap(m_h/60.0,6.0);
137  double S=clamp(m_s/100.0,0.0,1.0);
138  double V=clamp(m_v/100.0,0.0,1.0);
139 
140  // Convert to RGB.
141  if (S) {
142  int i=static_cast<int>(roundToZero(H));
143  double f=H-i;
144 
145  double p=V*(1-S);
146  double q=V*(1-S*f);
147  double t=V*(1-S*(1-f));
148 
149  switch (i) {
150  default: {
151  r=V; g=t; b=p;
152  break;
153  }
154  case 1: {
155  r=q; g=V; b=p;
156  break;
157  }
158  case 2: {
159  r=p; g=V; b=t;
160  break;
161  }
162  case 3: {
163  r=p; g=q; b=V;
164  break;
165  }
166  case 4: {
167  r=t; g=p; b=V;
168  break;
169  }
170  case 5: {
171  r=V; g=p; b=q;
172  break;
173  }
174  }
175  }
176  else {
177  // Achromatic case (gray), actually hue is undefined!
178  r=g=b=V;
179  }
180  }
181

242  static void RYBHueToRGBHue(double const h_ryb,double& h_rgb) {
243  ColorModelHSV hsv(h_ryb,100,100);
244 
245  // Abuse the HSV to RGB conversion: The amount of green for a given hue
246  // in RGB matches the amount of yellow for the same hue in RYB.
247  double r,y,b;
248  hsv.toRGB(r,y,b);
249 
250  // Convert RYB to RGB.
251  ColorModelRYB ryb(r,y,b);
252  double g;
253  ryb.toRGB(r,g,b);
254 
255  // Get the hue in RGB.
256  hsv.fromRGB(r,g,b);
257  h_rgb=hsv.getH();
258  }

263  static void RGBHueToRYBHue(double const h_rgb,double& h_ryb,double const a=0,double const b=360) {
264  // Guess an RYB hue ...
265  h_ryb=(a+b)*0.5;
266 
267  // ... and convert it to an RGB hue ...
268  double h_rgb_guessed;
269  RYBHueToRGBHue(h_ryb,h_rgb_guessed);
270 
271  // ... to see how closely we match the given RGB hue ...
272  double h_rgb_diff=h_rgb_guessed-h_rgb;
273  if (abs(h_rgb_diff)<=Numd::ZERO_TOLERANCE()) {
274  return;
275  }
276  // ... and adjust our guess accordingly, if required.
277  else if (h_rgb_diff<0) {
278  RGBHueToRYBHue(h_rgb,h_ryb,h_ryb,b);
279  }
280  else {
281  RGBHueToRYBHue(h_rgb,h_ryb,a,h_ryb);
282  }
283  }

301  void fromRGB(double const r,double const g,double const b) {
302  // Convert RGB to HSV.
303  ColorModelHSV hsv_rgb;
304  hsv_rgb.fromRGB(r,g,b);
305 
306  // Get the RYB hue for the RGB hue.
307  double h_ryb;
308  RGBHueToRYBHue(hsv_rgb.getH(),h_ryb);
309 
310  // Abuse the HSV to RGB conversion: The amount of green for a given hue
311  // in RGB matches the amount of yellow for the same hue in RYB.
312  ColorModelHSV hsv_ryb(h_ryb,100,100);
313  Tup3d ryb_hue_only;
314  hsv_ryb.toRGB(ryb_hue_only[0],ryb_hue_only[1],ryb_hue_only[2]);
315 
316  Tup3d white(0,0,0);
317  Tup3d black(1,1,1);
318  Tup3d ryb_with_saturation=lerp(white,ryb_hue_only,hsv_rgb.getS()/100.0);
319  Tup3d ryb=lerp(black,ryb_with_saturation,hsv_rgb.getV()/100.0);
320 
321  m_r=ryb[0];
322  m_y=ryb[1];
323  m_b=ryb[2];
324  }

328  void toRGB(double& r,double& g,double& b) const {
329  // Pair-wise interpolate RGB corner values along the B-axis of the RYB
330  // interpolation cube.
331  Tup3d b00=cubic(WHITE() , BLUE() , m_b);
332  Tup3d b10=cubic(YELLOW() , GREEN() , m_b);
333  Tup3d b01=cubic(RED() , PURPLE() , m_b);
334  Tup3d b11=cubic(ORANGE() , BLACK() , m_b);
335 
336  // Pair-wise interpolate RGB edge values along the Y-axis of the RYB
337  // interpolation cube.
338  Tup3d y0=cubic(b00,b10,m_y);
339  Tup3d y1=cubic(b01,b11,m_y);
340 
341  // Pair-wise interpolate RGB face values along the R-axis of the RYB
342  // interpolation cube.
343  Tup3d rgb=cubic(y0,y1,m_r);
344 
345  r=rgb[0];
346  g=rgb[1];
347  b=rgb[2];
348  }
