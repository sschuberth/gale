473  Color mixAdd(Color const& c,double s=0.5) const {
474  Color<N,double> mix=(Color<N,double>(*this)*(1-s)+Color<N,double>(c)*s)*2;
475 
476 // Warning C4127: Conditional expression is constant.
477 #pragma warning(disable:4127)
478 
479  if (N==4) {
480  // Undo the multiplication by 2 from above for the alpha channel.
481  mix.setA(mix.getA()*0.5);
482  }
483 
484 #pragma warning(default:4127)
485 
486  // Normalize the result to the unit color cube.
487  double max=mix.maxElement();
488  if (max>1.0) {
489  mix/=max;
490  }
491 
492  return Color(mix);
493  }

496  Color mixSub(Color const& c,double s=0.5) const {
