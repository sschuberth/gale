96  void onResize(int width,int height) {

98  static double const init_fov=math::Constd::PI()*0.25;
99 
100  double fov;
101 
102  if (width>=init_width) {
103  // Scale the FOV down in the same ration the width increased.
104  fov=static_cast<double>(init_width)/width*init_fov;
105  }
106  else {

108  fov=(init_fov-math::Constd::PI())/init_width*width+math::Constd::PI();
