104  m_stop=times(NULL);
105  if (m_stop==-1) {

109  // On modern CPUs with power-saving capabilities the clock ticks may
110  // vary, so get it each time here.
111  long ticks=sysconf(_SC_CLK_TCK);
112  if (ticks==-1) {
113  return false;
114  }
115 
116  m_offset+=m_stop-m_start;
117  seconds=static_cast<double>(m_offset)/ticks;
118 #elif defined G_OS_WINDOWS

127  return false;
128  }
129

131  m_offset+=m_stop.QuadPart-m_start.QuadPart;
132  seconds=static_cast<double>(m_offset)/frequency.QuadPart;
