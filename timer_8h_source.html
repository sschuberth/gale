102  bool stop(double& seconds) {
103 #ifdef G_OS_LINUX
104  m_stop=times(NULL);
105  if (m_stop==-1) {
106  return false;
107  }
108 
109  // On modern CPUs with power-saving capabilities the clock ticks may
110  // vary, so get it each time here.
111  long ticks=sysconf(_SC_CLK_TCK);
112  if (ticks==-1) {
113  return false;
114  }
115 
116  m_offset+=m_stop-m_start;
117  seconds=static_cast<double>(m_offset)/ticks;
118 #elif defined G_OS_WINDOWS
119  if (QueryPerformanceCounter(&m_stop)!=TRUE) {
120  return false;
121  }
122 
123  // On modern CPUs with power-saving capabilities the frequency may vary,
124  // so get it each time here.
125  LARGE_INTEGER frequency;
126  if (QueryPerformanceFrequency(&frequency)!=TRUE) {
127  return false;
128  }
129 
130  // Convert the counter difference to the number of seconds elapsed.
131  m_offset+=m_stop.QuadPart-m_start.QuadPart;
132  seconds=static_cast<double>(m_offset)/frequency.QuadPart;
133 #endif
134 
135  return true;
136  }
