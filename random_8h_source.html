103  g_uint32 random(g_uint32 const range) {
104  // Mask all bits starting from the MSB set in the range.
105  g_uint32 mask=range;
106  mask|=mask>>1;
107  mask|=mask>>2;
108  mask|=mask>>4;
109  mask|=mask>>8;
110  mask|=mask>>16;
111 
112  // Draw numbers until one in range [0,range] is found.
113  g_uint32 n;
114  do {
115  n=random()&mask;
116  } while (n>range);
117 
118  return n;
119  }
