586 G_INLINE FP32 convertHalfToFloat(FP16 h)
587 {
588  static FP32 const magic={(254-15)<<23};
589  static FP32 const was_infnan={(127+16)<<23};
590  FP32 o;
591 
592  // Exponent/mantissa bits.
593  o.u=(h.u&0x7fff)<<13;
594 
595  // Exponent adjustment.
596  o.f*=magic.f;
597 
598  // Make sure Inf/NaN survive.
599  if (o.f>=was_infnan.f)
600  o.u|=255<<23;
601 
602  // Sign bit.
603  o.u|=(h.u&0x8000)<<16;
604 
605  return o;
606 }

610 G_INLINE FP16 convertFloatToHalf(FP32 f)
611 {
612  FP32 infty={31<<23};
613  FP32 magic={15<<23};
614  FP16 o={0};
615 
616  unsigned int sign=f.sign;
617  f.sign=0;
618 
619  // If all exponent bits are set this is Inf or NaN.
620  if (f.exponent == 255)
621  {
622  // NaN->qNaN and Inf->Inf.
623  o.exponent=31;
624  o.mantissa=f.mantissa?0x200:0;
625  }
626  // This is a (de)normalized number or zero.
627  else {
628  // Make sure we don't get sticky bits.
629  f.u&=~0xfff;
630 
631  // Shift exponent down, denormalize if necessary.
632  f.f*=magic.f;
633 
634  // Rounding bias.
635  f.u+=0x1000;
636  if (f.u>infty.u) {
637  // Clamp to signed infinity if overflowed.
638  f.u=infty.u;
639  }
640 
641  // Take the bits!
642  o.u=f.u>>13;
643  }
644 
645  o.sign=sign;
646 
647  return o;
648 }

658 inline unsigned int countLeadingZeroBits(unsigned int x)
659 {
660 #ifdef G_COMP_GNUC
661 
662  if (x==0) {
663  return sizeof(x)*8;
664  }
665 
666  // At this point, we know there is at least one bit set.
667  return __builtin_clz(x);
668 
669 #elif defined(G_COMP_MSVC)
670 
671  DWORD index;
672  if (!_BitScanReverse(&index,x)) {
673  return sizeof(x)*8;
674  }
675 
676  unsigned int const MSB=sizeof(x)*8-1;
677  return index^MSB;
678 
679 #else // G_COMP_GNUC
680 
681  if (x==0) {
682  return sizeof(x)*8;
683  }
684 
685  unsigned int const MSB=sizeof(x)*8-1;
686 
687  // At this point, we know there is at least one bit set.
688  unsigned int count=0;
689  unsigned int const mask=1UL<<MSB;
690 
691  while ((x&mask)==0) {
692  ++count;
693  x<<=1UL;
694  }
695 
696  return count;
697 
698 #endif // G_COMP_GNUC
699 }

827 inline unsigned int ceilPow2(unsigned int const x)
828 {

830 
831  unsigned int result;
832  __asm__(
833  "decl %%edx\n\t"
834  "xorl %%eax,%%eax\n\t"
835  "bsrl %%edx,%%ecx\n\t"
836  "cmovzl %%eax,%%ecx\n\t"
837  "setnz %%al\n\t"

839  "shll %%cl,%%eax\n\t"
840  : "=a" (result) /* Output */
