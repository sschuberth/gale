108  static Matrix4 PerspectiveProjection(
109  int const view_width
110  , int const view_height
111  , double const fov=Constd::PI()*0.25
112  , double const clip_near=0.001
113  , double const clip_far=1000.0
114  )
115  {
116  // For details, see the gluPerspective() documentation in the OpenGL SDK.
117  double f=1.0/tan(fov*0.5);
118  double a=static_cast<double>(view_width)/view_height;
119 
120  double s=clip_near+clip_far;

342  Vector<3,T> mulMatVec(Vector<3,T> const& v) const {
343  // 16 scalar muls/divs, 12 scalar adds/subs.
344  T w=c0[3]*v[0] + c1[3]*v[1] + c2[3]*v[2] + c3[3];

526  static unsigned int const indices[]={
527  0, 0, 0, 4, 4, 8
528  , 5, 9, 13, 9, 13, 13

550  Matrix4 adjoint(T const a[6],T const b[6]) const {
551  Vec v[4];
552 
553  unsigned int o0=0;
554  unsigned int o1=1;
555  T const* f=b;
556 
557  for (int i=0;i<2;++i) {
558  v[o0]=Vec(
559  + c1[o1]*f[5] - c2[o1]*f[4] + c3[o1]*f[3]
560  , - c0[o1]*f[5] + c2[o1]*f[2] - c3[o1]*f[1]
561  , + c0[o1]*f[4] - c1[o1]*f[2] + c3[o1]*f[0]
562  , - c0[o1]*f[3] + c1[o1]*f[1] - c2[o1]*f[0]
563  );
564  v[o1]=Vec(
565  - c1[o0]*f[5] + c2[o0]*f[4] - c3[o0]*f[3]
566  , + c0[o0]*f[5] - c2[o0]*f[2] + c3[o0]*f[1]
567  , - c0[o0]*f[4] + c1[o0]*f[2] - c3[o0]*f[0]
568  , + c0[o0]*f[3] - c1[o0]*f[1] + c2[o0]*f[0]
569  );
570 
571  o0+=2;
572  o1+=2;
573  f=a;
574  }
575 
576  return Matrix4(v[0],v[1],v[2],v[3]);
577  }
