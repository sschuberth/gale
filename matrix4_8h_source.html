108  static Matrix4 PerspectiveProjection(
109  int const view_width
110  , int const view_height
111  , double const fov=Constd::PI()*0.25
112  , double const clip_near=0.001
113  , double const clip_far=1000.0

550  Matrix4 adjoint(T const a[6],T const b[6]) const {
551  Vec v[4];
552 
553  unsigned int o0=0;
554  unsigned int o1=1;
555  T const* f=b;
556 
557  for (int i=0;i<2;++i) {
558  v[o0]=Vec(
559  + c1[o1]*f[5] - c2[o1]*f[4] + c3[o1]*f[3]
560  , - c0[o1]*f[5] + c2[o1]*f[2] - c3[o1]*f[1]
561  , + c0[o1]*f[4] - c1[o1]*f[2] + c3[o1]*f[0]
562  , - c0[o1]*f[3] + c1[o1]*f[1] - c2[o1]*f[0]
563  );
564  v[o1]=Vec(
565  - c1[o0]*f[5] + c2[o0]*f[4] - c3[o0]*f[3]
566  , + c0[o0]*f[5] - c2[o0]*f[2] + c3[o0]*f[1]
567  , - c0[o0]*f[4] + c1[o0]*f[2] - c3[o0]*f[0]
