183  unsigned int processors() const {
184  unsigned int count=1;
185 
186 #ifdef G_OS_LINUX
187 
188  // Get the number of "online" processors. This counts processor cores,
189  // but not logical processors.
190  count=static_cast<unsigned int>(sysconf(_SC_NPROCESSORS_ONLN));
191 
192 #elif defined G_OS_WINDOWS
193 
194  SYSTEM_INFO info;
195  GetSystemInfo(&info);
196 
197  // This counts all logical processors including processor cores.
198  count=info.dwNumberOfProcessors;
199 
200  // Divide by the number of logical processors per processor core.
201  count/=logicalProcsPerCore();
202 
203 #endif
204 
205  // Divide by the number of processor cores per physical processor.
206  count/=coresPerPhysicalProc();
207 
208  return count<1?1:count;
209  }

212  unsigned int coresPerPhysicalProc() const {
213  unsigned int count=1;
214 
215  if (hasHTT()) {
216  if (isIntel()) {
217  // Get the maximum number of logical processors per physical processor.
218  unsigned int max_lpp=(m_00000001_ebx&0x00ff0000UL)>>16;
219 
220  // Get the maximum number of processor cores per physical processor.
221  unsigned int max_cpp=((m_00000004_eax&0xfc000000UL)>>26)+1;
222 
223  // Calculate the maximum number of logical processors per processor core.
224  unsigned int max_lpc=max_lpp/max_cpp;
225 
226  count=max_cpp/max_lpc;
227  }
228  else if (isAMD()) {
229  count=(m_80000008_ecx&0x000000ffUL)+1;
230  }
231  }
232 
233  return count<1?1:count;
234  }

237  unsigned int logicalProcsPerCore() const {
238  unsigned int count=1;
239 
240  if (hasHTT() && isIntel()) {
241  if (maxCPUIDStdFunc()>=0x0b) {
242 
243 #ifdef G_COMP_MSVC
244 
245  // Dirty trick to set the ECX register to 0 ("Count") as MSVC 8.0
246  // does not support inline assembly on the x86-64 architecture.
247  __stosd(reinterpret_cast<DWORD*>(&count),0,0);
248 
249  // Return the number of factory-configured logical processors
250  // at the thread level.
251  int info[4];
252  __cpuid(info,0x0000000b);
253  count=static_cast<unsigned int>(info[1])&0x0000ffff;
254 
255 #elif defined(G_COMP_GNUC) // G_COMP_MSVC
256 
257  unsigned int m_0000000b_ebx;
258 
259  __asm__(
260  "movl $0x0000000b,%%eax\n\t"
261  "xorl %%ecx,%%ecx\n\t"
262  "movl %%ebx,%%esi\n\t"
263  "cpuid\n\t"
264  "xchg %%ebx,%%esi\n\t"
265  : "=S" (m_0000000b_ebx) /* Output */
266  : /* Input */
267  : "%eax", "%ecx", "%edx", "cc" /* Clobber */
268  );
269 
270  count=m_0000000b_ebx&0x0000ffff;
271 
272 #endif
273 
274  }
275  else {
276  // Get the maximum number of logical processors per physical processor.
277  unsigned int max_lpp=(m_00000001_ebx&0x00ff0000UL)>>16;
278 
279  // Get the maximum number of processor cores per physical processor.
280  unsigned int max_cpp=((m_00000004_eax&0xfc000000UL)>>26)+1;
281 
282  // Calculate the maximum number of logical processors per processor core.
283  count=max_lpp/max_cpp;
284  }
285  }
286 
287  return count<1?1:count;
288  }
