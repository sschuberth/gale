292  void align(model::AABB const& box) {
293  float cath=math::max(box.getWidth(),box.getHeight())*0.5f;
294  float dist=cath/static_cast<float>(tan(m_fov*0.5))+box.getDepth()*0.5f;
295  math::Vec3f center=box.center();
296  math::HMat4f mat=math::HMat4f::Factory::LookAt(center,center+math::Vec3f(0,0,dist),math::Vec3f::Y());
297  setModelview(mat);
298  }
