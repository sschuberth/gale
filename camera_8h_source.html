292  void align(model::AABB const& box) {
293  float cath=math::max(box.getWidth(),box.getHeight())*0.5f;
294  float dist=cath/static_cast<float>(tan(m_fov*0.5))+box.getDepth()*0.5f;
295  math::Vec3f center=box.center();
296  math::HMat4f mat=math::HMat4f::Factory::LookAt(center,center+math::Vec3f(0,0,dist),math::Vec3f::Y());
297  setModelview(mat);
298  }

349  void rotate(math::Vec3f const& axis,double const angle) {
350  math::Quatf q(~axis,angle);
351 
352  // Normalize the vectors to avoid rounding errors.
353  m_modelview.setRightVector(~(q*m_modelview.getRightVector()));
354  m_modelview.setUpVector(~(q*m_modelview.getUpVector()));
355  m_modelview.setBackwardVector(~(q*m_modelview.getBackwardVector()));
356 
357  m_modelview_changed=true;
