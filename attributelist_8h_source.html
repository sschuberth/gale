105 
106  int pos=find(type);
107  if (pos>0) {
108  // Replace its value if the type already exists.
109  m_attributes[pos+1]=value;
110  return false;
111  }
112 
113  // Append the new attribute.
114  m_attributes[m_size++]=type;
115  m_attributes[m_size++]=value;
116  m_attributes[m_size]=0;
117 
118  return true;

123  int pos=find(type);
124  if (pos<0) {
125  // Return immediately if the type does not exist.
126  return false;
127  }
128 
129  // Remove the attribute by overwriting it with succeeding attributes.
130  while (pos+2<=m_size) {
131  m_attributes[pos]=m_attributes[pos+2];
132  ++pos;
133  }
