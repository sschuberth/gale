213  void setSize(int size) {
214  if (size<0) {
215  size=0;
216  }
217 
218  int i;
219 
220  // If we shrink in size, destroy abundant items.
221  for (i=size;i<m_size;++i) {
222  m_data[i].~T();

402  // Use binary search to find the item.

406  if (item>m_data[index]) {
