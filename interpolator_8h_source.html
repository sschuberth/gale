185  static float getInterval(int const n,float const s,bool const closed,int& i0,int& i1) {
186  G_ASSERT(n>=0)
187 
188  // Calculate the array indices from the position.
189  float i=(n+static_cast<int>(closed))*s;
190  int it=roundToZero(i);
191 
192  i0=(it>n)?0:it;
193  i1=(i0==n)?(closed?0:n):i0+1;
194 
195  return i-it;
196  }

200  template<class T>
201  static T evalPolynomial(T const& v0,T const& v1,T const& v2,T const& v3,signed char const (&w)[16],float const f,float const s) {
202  T a=f*(v0*w[ 0] + v1*w[ 1] + v2*w[ 2] + v3*w[ 3]);
203  T b=f*(v0*w[ 4] + v1*w[ 5] + v2*w[ 6] + v3*w[ 7]);
204  T c=f*(v0*w[ 8] + v1*w[ 9] + v2*w[10] + v3*w[11]);
205  T d=f*(v0*w[12] + v1*w[13] + v2*w[14] + v3*w[15]);
206 
207  return ((a*s + b)*s + c)*s + d;
208  }
