417  friend Quaternion slerp(Quaternion const& q,Quaternion const& r,float const s) {
418  T cosine=q.angleCosine(r);
419 
420  if (meta::OpCmpEqual::evaluate(cosine,T(1))) {
421  // If the quaternions are very close just interpolate linearly.
422  return nlerp(q,r,s);
423  }
424 
425  // Calculate the angle between q and r.
426  T angle=acos(cosine);
427 
428  // Calculate the angle between q and the interpolated result.
429  T theta=angle*s;
430 
431  return (
432  q * sin(angle-theta)
433  + r * sin( theta)
434  )/sin(angle);
435  }

510  void setMatrix(HMat const& m) {
511  T trace=m(0,0)+m(1,1)+m(2,2);
512 
513  if (trace>0) {
514  T r=sqrt(trace+1);
515  T s=T(0.5)/r;
516 
517  real = T(0.5)*r;
518  imag[0] = (m(2,1)-m(1,2))*s;
519  imag[1] = (m(0,2)-m(2,0))*s;
520  imag[2] = (m(1,0)-m(0,1))*s;
521  }
522  else {
523  unsigned int i=0;
524 
525  if (m(1,1)>m(0,0)) {
526  i=1;
527  }
528  if (m(2,2)>m(i,i)) {
529  i=2;
530  }
531 
532  unsigned int j=mod3(i+1);
533  unsigned int k=mod3(j+1);
534 
535  T r=sqrt(m(i,i)-m(j,j)-m(k,k)+1);
536  T s=T(0.5)/r;
537 
538  real = (m(k,j)-m(j,k))*s;
539  imag[i] = T(0.5)*r;
540  imag[j] = (m(j,i)+m(i,j))*s;
541  imag[k] = (m(k,i)+m(i,k))*s;
542  }
543  }

547  void getMatrix(HMat& m) const {
548  T x,y,z;
549  T wx,wy,wz,xx,xy,xz,yy,yz,zz;
550 
551  x=2*imag.getX();
552  y=2*imag.getY();
553  z=2*imag.getZ();
554 
555  wx=x*real;
556  wy=y*real;
557  wz=z*real;
558 
559  xx=x*imag.getX();
560  xy=y*imag.getX();
561  xz=z*imag.getX();
562 
563  yy=y*imag.getY();
564  yz=z*imag.getY();
565 
566  zz=z*imag.getZ();
567 
568  m.c0=Vec(T(1-(yy+zz)), T( (xy+wz)), T( (xz-wy)));
569  m.c1=Vec(T( (xy-wz)), T(1-(xx+zz)), T( (yz+wx)));
570  m.c2=Vec(T( (xz+wy)), T( (yz-wx)), T(1-(xx+yy)));
571  m.c3=Vec::ZERO();
572  }

576  Quaternion tangent(Quaternion const& a,Quaternion const& b) const {
577  // For normalized quaternions, the conjugate equals the inverse.
578  Quaternion inv=*this;
579  inv.conjugate();
