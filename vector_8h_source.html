301  double orientedAngle(Vector const& v,Vector const& r) const {
302  Vector tn=~(*this),vn=~v;
303 
304  // Calculate the normal between the two angle vectors and project the
305  // reference vector onto it.
306  Vector n=(*this)^v;
307  T d=r%n;
308 
309  if (d==0 && tn+vn==Vector::ZERO()) {
310  // If the two normalized angle vectors compensate each other,
311  // they point to opposite directions, and the angle should
312  // be exactly 180 degrees.
313  return Constd::PI();
314  }
315 
316  // Calculate the unoriented angle between the two angle vectors.
317  double a=acos(static_cast<double>(tn%vn));
318 
319  if (d<0) {
320  // If d is negative, r projects onto the negative direction of n,
321  // i.e. it points to the other half space with respect to the
322  // plane spanned by the two angle vectors, and the angle amounts
323  // to what is missing to a full circle.
324  a=2*Constd::PI()-a;
325  }
326 
327  return a;
328  }

336  void crossWith(Vector const& v) {
337  // 6 scalar muls/divs, 3 scalar adds/subs.
338  G_ASSERT(N==3)
339  T x = getY()*v.getZ() - getZ()*v.getY();
340  T y = getZ()*v.getX() - getX()*v.getZ();
341  T z = getX()*v.getY() - getY()*v.getX();
342  set(x,y,z);
343  }
