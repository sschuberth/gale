150  static HMatrix4 LookAt(Vec const& target,Vec const& position,Vec up) {
151  Vec backward=~(position-target),right;
152 
153  if (backward.isCollinear(up)) {
154  up=backward.orthoVector();
155  }
156 
157  right=~(up^backward);
158  up=backward^right;

386  friend HMatrix4 operator*(HMatrix4 const& m,HMatrix4 const& n) {
387  Vec c0,c1,c2,c3;
388 
389  // 36 scalar muls/divs, 27 scalar adds/subs (includes translation).
390  for (int row=2;row>=0;--row) {
391  int col1=row+4,col2=row+8,col3=row+12;
392  c0[row] = m[row]*n[ 0] + m[col1]*n[ 1] + m[col2]*n[ 2];
393  c1[row] = m[row]*n[ 4] + m[col1]*n[ 5] + m[col2]*n[ 6];
394  c2[row] = m[row]*n[ 8] + m[col1]*n[ 9] + m[col2]*n[10];
395  c3[row] = m[row]*n[12] + m[col1]*n[13] + m[col2]*n[14] + m[col3];
396  }
397 
398  return HMatrix4(c0,c1,c2,c3);
399  }

513  T determinant() const {
514  T a0 = c0[0]*c1[1] - c1[0]*c0[1];
515  T a1 = c0[0]*c2[1] - c2[0]*c0[1];
516  T a3 = c1[0]*c2[1] - c2[0]*c1[1];
517 
518  T const& b2 = c0[2];
519  T const& b4 = c1[2];
520  T const& b5 = c2[2];
521 
522  return a0*b5 - a1*b4 + a3*b2;
523  }

527  void orthonormalize() {
528  c0.normalize();
529  c1=~(c1-(c1%c0)*c0);
530 
531  // This is cheaper than what the real modified Gram-Schmidt algorithm does. Also see
532  // https://fgiesen.wordpress.com/2013/06/02/modified-gram-schmidt-orthogonalization/.
533  c2=c0^c1;
534  }

537  void transpose() {
538  T tmp;
539  tmp=(*this)(1,0); (*this)(1,0)=(*this)(0,1); (*this)(0,1)=tmp;
540  tmp=(*this)(2,0); (*this)(2,0)=(*this)(0,2); (*this)(0,2)=tmp;
541  tmp=(*this)(2,1); (*this)(2,1)=(*this)(1,2); (*this)(1,2)=tmp;
542  }

546  void invert(bool* const result=NULL) {
547  bool valid=abs(determinant())>Numerics<T>::ZERO_TOLERANCE();
548 
549  if (result) {
550  *result=valid;
551  }
552 
553  if (valid) {
554  // Invert any scaling.
555  c0/=c0.length2();
556  c1/=c1.length2();
557  c2/=c2.length2();
558 
559  // Invert any translation.
560  c3=Vec(-c3%c0,-c3%c1,-c3%c2);
561 
562  // Invert any rotation.
563  transpose();
564  }
