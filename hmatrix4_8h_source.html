534  void transpose() {
535  T tmp;
536  tmp=(*this)(1,0); (*this)(1,0)=(*this)(0,1); (*this)(0,1)=tmp;
537  tmp=(*this)(2,0); (*this)(2,0)=(*this)(0,2); (*this)(0,2)=tmp;

543  void invert(bool* const result=NULL) {
544  bool valid=abs(determinant())>Numerics<T>::ZERO_TOLERANCE();
545 
546  if (result) {
547  *result=valid;
548  }
549 
550  if (valid) {
551  // Invert any scaling.
552  c0/=c0.length2();
553  c1/=c1.length2();
554  c2/=c2.length2();
555 
556  // Invert any translation.
557  c3=Vec(-c3%c0,-c3%c1,-c3%c2);
558 
559  // Invert any rotation.
560  transpose();
561  }
