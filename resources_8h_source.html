38 {
39 #ifndef GALE_TINY_CODE
40  // Check ICONDIR structure entries, see
41  // <http://msdn.microsoft.com/en-us/library/ms997538.aspx>.
42  LPWORD icondir=LPWORD(data);
43  if (!icondir || icondir[0]!=0 || icondir[1]!=1 || icondir[2]<=index) {
44  return NULL;
45  }
46 #endif
47 
48  // Skip the ICONDIR and preceding ICONDIRENTRY structures.
49  LPDWORD icondirentry=LPDWORD(data+6+index*16);
50
